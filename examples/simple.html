Example with default settings:

            
$(function(){
    $('#slugme1').slugIt({
        output: '.slug1'
    });
});
id: slugme1
class: slug1
class: slug1


Example with custom settings:

            
$(function(){
    $('#slugme2').slugIt({
        output: '.slug2',
        separator: '_',
        map: {
            '☂': 'umbrella'
        }
    });
});
id: slugme2
class: slug2
class: slug2
